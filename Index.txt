  let confettiActive = false;

function launchConfetti(duration = 2000) {
  const canvas = document.getElementById('confettiCanvas');
  canvas.style.display = 'block';
  const ctx = canvas.getContext('2d');
  const dpr = window.devicePixelRatio || 1;

  canvas.width = window.innerWidth * dpr;
  canvas.height = window.innerHeight * dpr;
  ctx.scale(dpr, dpr);

  const colors = ['#ff6b6b','#feca57','#54a0ff','#1dd1a1','#5f27cd'];
  const pieces = [];

  for (let i = 0; i < 140; i++) {
    pieces.push({
      x: Math.random() * window.innerWidth,
      y: Math.random() * -window.innerHeight,
      size: 6 + Math.random() * 6,
      color: colors[Math.floor(Math.random() * colors.length)],
      speed: 2 + Math.random() * 3,
      tilt: Math.random() * 10
    });
  }

  confettiActive = true;

  function draw() {
    ctx.clearRect(0, 0, canvas.width, canvas.height);

    pieces.forEach(p => {
      ctx.beginPath();
      ctx.fillStyle = p.color;
      ctx.fillRect(p.x, p.y, p.size, p.size / 2);
      p.y += p.speed;
      p.x += Math.sin(p.tilt);
    });

    if (confettiActive) requestAnimationFrame(draw);
  }

  draw();

setTimeout(() => {
  confettiActive = false;
  ctx.clearRect(0, 0, canvas.width, canvas.height);
  canvas.style.display = 'none';
}, duration);
}

  grid.className = "jobs-grid";

  places.forEach(p => {

    const box = document.createElement("div");
    box.className = "job-box";

    box.innerHTML = `
      <div class="job-name">${p.name}</div>

  return gCard;
}
function setLoadingState(isLoading, text = "جاري التحميل…") {
  document.querySelectorAll('.download-btn').forEach(btn => {
    if (isLoading) {
      btn.dataset.oldText = btn.textContent;
      btn.textContent = text;
      btn.classList.add('loading');
      btn.disabled = true;
    } else {
      btn.textContent = btn.dataset.oldText || 'تنزيل ملف الإكسل';
      btn.classList.remove('loading');
      btn.disabled = false;
    }
  });
}

function startLoadingBar() {
  const overlay = document.getElementById('loadingOverlay');
  const track = document.querySelector('.loading-track');
  const bar = document.querySelector('.loading-bar');
  const horse = document.querySelector('.horse');

  overlay.style.display = 'flex';
  bar.style.width = '0%';
  horse.style.right = '0px'; // يبدأ من اليمين فعلًا

  let progress = 0;

  function animate() {
    if (progress < 90) {

let speed;
if (progress < 40) speed = 0.6;
else if (progress < 70) speed = 0.35;
else if (progress < 85) speed = 0.18;
else speed = 0.06; // 👈 حركة بطيئة مستمرة قرب 90


      progress = Math.min(progress + speed, 90);
      bar.style.width = progress + '%';

      // ✅ حساب موضع الحصان من عرض البار الحقيقي
      const barWidth = bar.getBoundingClientRect().width;
      horse.style.right = barWidth + 'px';

      requestAnimationFrame(animate);
    }
  }

  requestAnimationFrame(animate);
}




function finishLoadingBar() {
  const bar = document.querySelector('.loading-bar');
  const horse = document.querySelector('.horse');
  const overlay = document.getElementById('loadingOverlay');

  let progress = parseFloat(bar.style.width) || 90;

  function complete() {
    if (progress < 100) {

      // سرعة إكمال ناعمة
      progress = Math.min(progress + 0.8, 100);
      bar.style.width = progress + '%';

      // الحصان مربوط بعرض البار الحقيقي
      const barWidth = bar.getBoundingClientRect().width;
      horse.style.right = barWidth + 'px';

      requestAnimationFrame(complete);

    } else {
      // اختفاء اللودر بعد اكتمال 100%
      setTimeout(() => {
        overlay.style.display = 'none';
      }, 250);
    }
  }

  requestAnimationFrame(complete);
}



function showExcelLoading(text = "جاري تحميل ملف الأكسل...") {
  const overlay = document.getElementById("excelLoadingOverlay");
  if (!overlay) return;
  overlay.querySelector("div:last-child").textContent = text;
  overlay.style.display = "flex";
}

function hideExcelLoading() {
  const overlay = document.getElementById("excelLoadingOverlay");
  if (!overlay) return;
  overlay.style.display = "none";
}

/* زر تحميل الاكسل: ينادي دالة Apps Script الجديدة */
function downloadWorkforceExcel(){
  showExcelLoading("جاري تحميل ملف الأكسل...");
  setLoadingState(true);

  google.script.run
    .withFailureHandler(err=>{
      hideExcelLoading();
      setLoadingState(false);
      alert('حدث خطأ أثناء إنشاء الملف');
    })
    .withSuccessHandler(url=>{
      if(url) window.open(url, '_blank');

      setTimeout(() => {
        hideExcelLoading();
        setLoadingState(false);
      }, 800);
    })
    .exportWorkforceExcel();
}
